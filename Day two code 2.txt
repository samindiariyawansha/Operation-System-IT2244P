// create a file
C:\Users\Samindi Ariyawansha>dir >abc.txt


// Adds the Read-only attribute to abc.txt
C:\Users\Samindi Ariyawansha>ATTRIB +R abc.txt


// Removes the Read-only attribute from abc.txt
C:\Users\Samindi Ariyawansha>ATTRIB -R abc.txt


// Adds the Hidden attribute to abc.txt
C:\Users\Samindi Ariyawansha>ATTRIB +H abc.txt


// Removes the Hidden attribute to abc.txt
C:\Users\Samindi Ariyawansha>ATTRIB -H abc.txt
