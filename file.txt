#include <stdio.h>
#include <unistd.h>

int main() {
    int n = fork();

    if (n == 0) { 
        printf("\nI am a child process.");
        printf("\nMy ID: %d", getpid());
        printf("\nMy Parent ID: %d\n", getppid());
    } else { 
        printf("\nI am a parent process.");
        printf("\nMy ID: %d\n", getpid());
    }

    return 0;
}

//print numbers from one to and , 1 to 5 should be print by child process and from 6 to 10 should be parent process, calculate summation of thease numbers


#include <stdio.h>
#include <unistd.h>

int main() {
    int n = fork();

    if (n == 0) { 
        int sum = 0;
		
        for (int i = 1; i <= 5; i++) {
		sum = i; 
		printf("%d ", i);
		}
        printf("\nChild sum: %d\n", sum);
		
		
    } else {
    
        int sum = 0;
		
        for (int i = 6; i <= 10; i++)
		{
			sum += i;
			printf("%d ", i);
		}
			printf("\nParent sum: %d\nTotal sum: %d\n", sum, sum + 15);
    }

    return 0;
}
